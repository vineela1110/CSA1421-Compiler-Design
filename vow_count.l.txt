%{
#include <stdio.h>
int vowels = 0;
int consonants = 0;
%}

%%
[aeiouAEIOU]    { vowels++; }
[a-zA-Z]        { consonants++; }
\n              { return 0; }
.               { /* ignore other characters */ }
%%

int yywrap(void) {
    return 1;
}

int main() {
    printf("Enter a string: ");
    yylex();
    printf("\nVowels = %d", vowels);
    printf("\nConsonants = %d\n", consonants);
    return 0;
}
